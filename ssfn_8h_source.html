355  /* parse character table */
356  static uint8_t *_ssfn_c(const ssfn_font_t *font, const char *str, int *len, uint32_t *unicode)
357  {
358  uint32_t i, j, u = -1U;
359  uint16_t *l;
360  uint8_t *ptr, *s;
361 
362  *len = 0; *unicode = 0;
363  if(!font || !font->characters_offs || !str || !*str) return NULL;
364 
365  if(font->ligature_offs) {
366  for(l = (uint16_t*)((uint8_t*)font + font->ligature_offs), i = 0; l[i] && u == -1U; i++) {
367  for(ptr = (uint8_t*)font + l[i], s = (uint8_t*)str; *ptr && *ptr == *s; ptr++, s++);
368  if(!*ptr) { u = SSFN_LIG_FIRST + i; break; }
369  }
370  }
371  if(u == -1U) {
372  /* inline ssfn_utf8 to workaround -O2 bug in gcc 11.1 */
373  s = (uint8_t*)str; u = *s;
374  if((*s & 128) != 0) {
375  if(!(*s & 32)) { u = ((*s & 0x1F)<<6)|(*(s+1) & 0x3F); s++; } else
376  if(!(*s & 16)) { u = ((*s & 0xF)<<12)|((*(s+1) & 0x3F)<<6)|(*(s+2) & 0x3F); s += 2; } else
377  if(!(*s & 8)) { u = ((*s & 0x7)<<18)|((*(s+1) & 0x3F)<<12)|((*(s+2) & 0x3F)<<6)|(*(s+3) & 0x3F); s += 3; }
378  else u = 0;
379  }
380  s++;
381  }
382  *len = (int)(s - (uint8_t*)str);
383  *unicode = u;
384  for(ptr = (uint8_t*)font + font->characters_offs, i = 0; i < 0x110000; i++) {
385  if(ptr[0] == 0xFF) { i += 65535; ptr++; }
386  else if((ptr[0] & 0xC0) == 0xC0) { j = (((ptr[0] & 0x3F) << 8) | ptr[1]); i += j; ptr += 2; }
387  else if((ptr[0] & 0xC0) == 0x80) { j = (ptr[0] & 0x3F); i += j; ptr++; }
388  else {
389  if(i == u) return ptr;
390  ptr += 6 + ptr[1] * (ptr[0] & 0x40 ? 6 : 5);
391  }
392  }
393  return NULL;
394  }

488  static int _ssfn__zbuild_huffman(_ssfn__zhuffman *z, unsigned char *sizelist, int num)
489  {
490  int i,k=0;
491  int code, next_code[16], sizes[17];
492 
493  SSFN_memset(sizes, 0, sizeof(sizes));
494  SSFN_memset(z->fast, 0, sizeof(z->fast));
495  for (i=0; i < num; ++i)
496  ++sizes[sizelist[i]];
497  sizes[0] = 0;
498  for (i=1; i < 16; ++i)
499  if (sizes[i] > (1 << i))
500  return 0;
501  code = 0;
502  for (i=1; i < 16; ++i) {
503  next_code[i] = code;
504  z->firstcode[i] = (uint16_t) code;
505  z->firstsymbol[i] = (uint16_t) k;
506  code = (code + sizes[i]);
507  if (sizes[i])
508  if (code-1 >= (1 << i)) return 0;
509  z->maxcode[i] = code << (16-i);
510  code <<= 1;
511  k += sizes[i];
512  }
513  z->maxcode[16] = 0x10000;
514  for (i=0; i < num; ++i) {
515  int s = sizelist[i];
516  if (s) {
517  int c = next_code[s] - z->firstcode[s] + z->firstsymbol[s];
518  uint16_t fastv = (uint16_t) ((s << 9) | i);
519  z->size [c] = (unsigned char) s;
520  z->value[c] = (uint16_t) i;
521  if (s <= SSFN__ZFAST_BITS) {
522  int j = _ssfn__bit_reverse(next_code[s],s);
523  while (j < (1 << SSFN__ZFAST_BITS)) {
524  z->fast[j] = fastv;
525  j += (1 << s);
526  }
527  }
528  ++next_code[s];
529  }
530  }
531  return 1;
532  }
